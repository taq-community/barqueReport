\newcommand{\makecover}{
  \begin{titlepage}
    \singlespacing  % Use single spacing for title page
    % Background image
    $if(cover-image)$
    \AddToShipoutPictureBG*{%
      \AtPageUpperLeft{%
        \raisebox{-\height}{%
          \includegraphics[width=\paperwidth,height=\paperheight,keepaspectratio=false]{$cover-image$}%
        }%
      }%
    }
    $endif$

    \vspace*{0.1\textheight}

    \begin{center}
      % Title in white
      {\color{white}\LARGE \bfseries $title$ \par}
      % Subtitle in white
      $if(subtitle)$
      {\color{white}\LARGE $subtitle$ \par}
      $endif$

      \vspace{1.5cm}

      $if(client-name)$
      {\color{white} \normalsize \textbf{Pour } \par}
      {\color{white} \normalsize $client-name$}
      $endif$

      \vspace{1.5cm}

      {\color{white} \normalsize \textbf{Réalisé par} \par}
      {\color{white}Laboratoire de Valérie Langlois (Prof., Ph.D.) en écotoxicogénomique,\\ ADN/ARN environnemental et perturbation endocrinienne \\ INRS, 490, Rue de la Couronne, Québec, Québec G1K 9A9\par}

      \vspace{1cm}

      % Authorship sections
      $if(extraction-author)$
      {\color{white}\normalsize \textbf{Extraction d'ADN des échantillons} \par}
      {\color{white}\normalsize $extraction-author$}
      \vspace{0.5cm}
      $endif$


      $if(lab-author)$
      {\color{white}\normalsize \textbf{Analyse métagénomique en laboratoire} \par}
      {\color{white}\normalsize $lab-author$ \par}
      \vspace{0.5cm}
      $endif$

      $if(bioinfo-author)$
      {\color{white}\normalsize \textbf{Analyses bio-informatiques} \par}
      {\color{white}\normalsize $bioinfo-author$ \par}
      \vspace{0.5cm}
      $endif$

      $if(report-author)$
      {\color{white}\normalsize \textbf{Rédaction du rapport} \par}
      {\color{white}\normalsize $report-author$}
      \vspace{0.5cm}
      $endif$

      \vspace{1cm}
    \end{center}
  \end{titlepage}
  \onehalfspacing  % Resume 1.5 spacing after title page
  \clearpage
}
